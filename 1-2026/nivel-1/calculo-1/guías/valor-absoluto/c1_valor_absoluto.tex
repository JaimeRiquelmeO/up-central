\begin{ejercicios}

\subsection*{I. Valor Absoluto (dificultad ascendente)}

1.1) \textbf{Resuelva la inecuación} \hfill\textit{(Básico)}
$$|x - 2| < 3$$
\medskip

1.2) \textbf{Resuelva la inecuación} \hfill\textit{(Básico)}
$$|2x + 1| \geq 5$$
\medskip

1.3) \textbf{Encuentre el conjunto solución} \hfill\textit{(Intermedio)}
$$|x^2 - 4| \leq 5$$
\medskip

1.4) \textbf{Resuelva la siguiente inecuación} \hfill\textit{(Intermedio)}
$$|x - 1| + |x - 3| < 4$$
\medskip

1.5) \textbf{Determine el conjunto solución} \hfill\textit{(Intermedio-Alto)}
$$\left|\frac{x - 1}{x + 2}\right| \leq 1$$
\medskip

1.6) \textbf{Resuelva la inecuación} \hfill\textit{(Avanzado)}
$$|x^2 - 2x| < 3$$
\medskip

1.7) \textbf{Encuentre el conjunto solución, con $x \neq 0$} \hfill\textit{(Avanzado)}
$$\left|x + \frac{1}{x}\right| > 3$$
\medskip

1.8) \textbf{Resuelva la siguiente inecuación} \hfill\textit{(Intermedio)}
$$|x + 3| - |x - 1| > 2$$
\medskip

1.9) \textbf{Encuentre el conjunto solución} \hfill\textit{(Intermedio)}
$$|x^2 - x - 6| < 2$$
\medskip

1.10) \textbf{Resuelva la siguiente inecuación} \hfill\textit{(Intermedio-Alto)}
$$\frac{|x - 2|}{x^2 + 1} < 1$$
\medskip

1.11) \textbf{Determine el conjunto solución} \hfill\textit{(Avanzado)}
$$|x^2 - 4x + 3| \leq |x - 1|$$
\medskip

1.12) \textbf{Resuelva la siguiente inecuación} \hfill\textit{(Avanzado)}
$$2|x + 1| - |x - 3| > x + 5$$
\vspace{3cm}
\newpage

\end{ejercicios}
